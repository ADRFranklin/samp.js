/** Cached name

/**
 * Checks if player is valid
 * @type {Boolean}
 */
	get isValid(){
		return (this.id < 65535);
	}

/**
 * Get player name | GetPlayerName(playerid)

	get name(){
		this.name_ = GetPlayerName(this.id);
		return this.name_; 		
	}

/**

	}

/**
 * Get Player Position | GetPlayerPos(playerid)
 * @type {Object(x:Number,y:Number,z:Number,a:Number)}
 */ 
	get pos(){
		var tmpp = GetPlayerPos(this.id);
		var tmpa = this.facingAngle;
		this.pos_ = {x: tmpp.x, y: tmpp.y, z: tmpp.z, a: tmpa };
		return this.pos_;
	}
/**
 * Set Player Position. Takes an Object or an Array | SetPlayerPos(playerid, x, y, z )
 * @type {Object(x:Number,y:Number,z:Number,[a:Number])}
 */
	set pos(pos){
		if(Array.isArray(pos)){
			if(pos.length > 3) this.facingAngle = pos[3];
			this.pos_.x = pos[0];
			this.pos_.y = pos[1];
			this.pos_.z = pos[2];
		} else {
			if(pos.hasOwnProperty('a')) this.facingAngle = pos.a;
			this.pos_.x = pos.x;
			this.pos_.y = pos.y;
			this.pos_.z = pos.z;
		}
		SetPlayerPos(this.id,this.pos_.x,this.pos_.y,this.pos_.z);
	}
