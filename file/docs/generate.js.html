    var fs = require('fs');
var path = require('path');


var src = ['../javascript', 'E:/samp-server/js/include'];
var out = 'docs/js/include';



for(var dir in src ){
	parseDirectory(src[dir]);
}

function parseDirectory(directory){
	console.log("Parsing Directory "+directory);
	fs.readdir(directory, function(err,dirs){
		dirs.forEach(function(file){
			file = path.resolve(directory,file);
			fs.stat(file, function(err,stat){
				if(stat && stat.isDirectory()){
					console.log("Parsing Directory: "+file);
					parseDirectory(file);
					
				} else {
					if(path.extname(file) == '.js'){
						console.log("Parsing File: "+file);
						parseFile(file);	
					}
				}
			});
		});
	});
}

function parseFile(file){
	fs.readFile(file, {encoding: 'utf8'}, function(err,data){
		var lines = data.split('\r\n');
		var reg = new RegExp(/^class (.*)$/);
		var reg2 = new RegExp(/^function (.*)$/);
		var reg3 = new RegExp(/^const (.*)$/);
		for(var i in lines){
			if(reg.test(lines[i])){
				lines[i] = 'export '+lines[i];
				break;
			}
			else if(reg2.test(lines[i])){
			//	console.log("Match 2: "+file);
				lines[i] = 'export '+lines[i];	
			} else if(reg3.test(lines[i])){
				lines[i] = 'export '+lines[i];	
			}
		}
		
		fs.writeFile(out+'/'+path.basename(file), lines.join('\r\n'), function(err){});
		
	});
}
